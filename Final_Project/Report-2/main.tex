	اعتراضات دی‌ماه ۱۴۰۴ در ایران، بستری مهم برای مطالعه پویایی‌های شبکه‌های اجتماعی به عنوان بازتابی از منازعات اجتماعی و سیاسی فراهم آورد. نظریه تعادل ساختاری (که به نظریه هایدر نیز معروف است) چارچوبی قدرتمند برای تحلیل چگونگی توزیع روابط مثبت (ائتلاف) و منفی (تضاد) در یک شبکه اجتماعی ارائه می‌دهد. این نظریه بیان می‌کند که شبکه‌های پایدار تمایل دارند به گونه‌ای شکل گیرند که سه‌گانه‌های (Triads) تشکیل‌دهنده آن‌ها متعادل باشند؛ به این معنا که "دشمن دشمن، دوست است" و "دوست دوست، دوست است". در این گزارش، با بهره‌گیری از خروجی‌های کد تحلیلی که بر روی داده‌های سه پلتفرم اصلی (تلگرام، اینستاگرام و توییتر) اجرا شده، به واکاوی میزان تعادل ساختاری، شناسایی خوشه‌های اصلی (شامل هسته مرکزی موافق نظام و گروه‌های اپوزیسیون) و تحلیل روابط داخلی و خارجی این خوشه‌ها می‌پردازیم.
	
	\section{روش‌شناسی تحلیل}
	تحلیل حاضر بر دو پایه اصلی استوار است:

	\section{تحلیل تعادل ساختاری: شبکه‌ای در تنش}
	داده‌های استخراج‌شده از سه پلتفرم، تصویری از یک اکوسیستم شبکه‌ای ناپایدار را ترسیم می‌کند. جدول \ref{tab:triads} خلاصه‌ای از این داده‌ها را نشان می‌دهد.
	
	\begin{table}[h!]
		\centering
		\caption{تعداد سه‌گانه‌های متعادل و نامتعادل در پلتفرم‌های مختلف}
		\label{tab:triads}
		\begin{tabular}{@{}lccc@{}}
			\toprule
			\textbf{پلتفرم} & \textbf{سه‌گانه متعادل} & \textbf{سه‌گانه نامتعادل} & \textbf{درصد نامتعادل} \\ \midrule
			تلگرام (Telegram)   & 453                    & 723                      & 61.5\% \\

	نسبت بالای سه‌گانه‌های نامتعادل در تلگرام (بیش از ۶۱٪) و توییتر (بیش از ۶۴٪) نشان‌دهنده سطح بالایی از تنش و پیکربندی ناپایدار روابط است. از منظر نظریه تعادل ساختاری، چنین شبکه‌ای مستعد تغییر و بازآرایی روابط است. این ناپایداری را می‌توان به عوامل زیر نسبت داد:
	
	\begin{itemize}
		\item \textbf{شکاف‌های درون اپوزیسیون:} حضور همزمان گروه‌های مختلف اپوزیسیون (سلطنت‌طلب، کرد، ترک، بلوچ) با ایدئولوژی‌ها و گاه منافع متضاد، منجر به شکل‌گیری روابط منفی درون این ابرخوشه می‌شود. به عنوان مثال، یک حساب سلطنت‌طلب ممکن است با یک حساب کرد که خواستار خودمختاری بیشتر است، رابطه منفی (تضاد) داشته باشد، در حالی که هر دو با حکومت مرکزی رابطه منفی دارند. این پیکربندی، یک سه‌گانه نامتعادل کلاسیک است (دشمن دشمن من، دوست من نیست).
		\item \textbf{راهبردهای ارتباطی هسته مرکزی:} "خوشه مرکزی" (Core Cluster) که نماینده طیف موافق نظام است، ممکن است با برخی گروه‌های اپوزیسیون (مثلاً گروه‌های ملی‌گرای غیرخشونت‌طلب) رابطه خنثی یا حتی مثبت محدودی داشته باشد تا شکاف‌های درون اپوزیسیون را تعمیق بخشد. این راهبرد "تفرقه بینداز و حکومت کن" به طور مستقیم به تولید سه‌گانه‌های نامتعادل دامن می‌زند. تحقیقات پیشین نیز نشان داده است که چگونه بازیگران دولتی از تاکتیک‌های محاسباتی برای قطبی‌سازی فضا و تضعیف جنبش‌های اجتماعی استفاده می‌کنند [citation:9].
		\item \textbf{کمپین‌های هماهنگ و شایعات:} همانطور که در خروجی کد برای پلتفرم‌ها به بررسی "هماهنگی محتوا" (Coordination Heatmaps) اشاره شده، گروه‌های مختلف برای پیشبرد روایت‌های خود (مانند شایعات مربوط به کمبود سوخت یا سقوط پادگان‌ها) دست به اقدام هماهنگ می‌زنند. این هماهنگی، وفاداری‌های درون‌گروهی را تقویت و همزمان شکاف با گروه‌های رقیب را عمیق‌تر می‌کند.
	\end{itemize}
	
	\section{ریخت‌شناسی خوشه‌ها: از هسته مرکزی تا خرده‌گروه‌های اپوزیسیون}
	تحلیل جوامع، ساختار قطبی‌شده اما درون‌اً شکننده شبکه‌های اجتماعی ایران را نمایان می‌سازد. در هر دو پلتفرم تلگرام و توییتر، الگوی مشابهی از یک "خوشه مرکزی" در مقابل "خرده‌گروه‌های اپوزیسیون" دیده می‌شود.
	
	\subsection{تحلیل خوشه‌ها در تلگرام}
	نتایج حاصل از تلگرام (جدول \ref{tab:communities_telegram}) یک ساختار جالب را نشان می‌دهد:
	
	\begin{table}[h!]
		\centering
		\caption{خرده‌گروه‌های شناسایی‌شده در تلگرام}

			\toprule
			\textbf{شناسه جامعه} & \textbf{اندازه} & \textbf{اهمیت میانگین} & \textbf{نوع خوشه} \\ \midrule
			0                    & 3                & 1.64                   & خوشه مرکزی (Core Cluster) \\
			1                    & 2                & 1.30                   & خرده‌گروه اپوزیسیون \\

			3                    & 4                & 1.08                   & خرده‌گروه اپوزیسیون \\
			4                    & 119              & 2.53                   & خرده‌گروه اپوزیسیون \\ \bottomrule

	\begin{itemize}
		\item \textbf{خوشه مرکزی (جامعه ۰):} با وجود اندازه بسیار کوچک (۳ عضو)، از اهمیت میانگین قابل توجهی (۱.۶۴) برخوردار است. این یافته حیاتی است. این خوشه احتمالاً شامل حساب‌های کلیدی و بسیار فعال موافق نظام (مانند خبرگزاری‌های رسمی، روحانیون تأثیرگذار یا شخصیت�های سیاسی) است که وزن بالایی در شبکه دارند و به عنوان "گره‌های نافذ" (Influential Nodes) عمل می‌کنند. این هسته کوچک اما مقتدر، توانایی بالایی در روایت‌سازی و جهت‌دهی به افکار عمومی دارد [citation:7].
		\item \textbf{جامعه ۴ (اپوزیسیون بزرگ):} این خوشه با ۱۱۹ عضو، بزرگترین جامعه اپوزیسیون است و بالاترین میزان اهمیت میانگین (۲.۵۳) را در کل شبکه داراست. این موضوع نشان می‌دهد که بدنه اصلی مخالفان نه تنها از نظر تعداد، بلکه از نظر شدت فعالیت و تأثیرگذاری نیز وضعیت بالایی دارند. با این حال، تنوع درونی این خوشه بزرگ (شامل سلطنت‌طلبان، کردها، ترک‌ها و بلوچ‌ها) منبع اصلی نامتعادلی در شبکه است.
		\item \textbf{خرده‌گروه‌های کوچک اپوزیسیون (جوامع ۱، ۲ و ۳):} حضور این خرده‌گروه‌ها با اندازه‌های متفاوت (۲ تا ۱۰ عضو) و اهمیت میانگین متغیر (از ۱.۰۸ تا ۱.۹۵)، گویای وجود شکاف‌های هویتی و ایدئولوژیک درون اپوزیسیون است. این جوامع احتمالاً نماینده زیرشاخه‌های قومی یا ایدئولوژیک خاصی هستند که با یکدیگر و با خوشه اصلی اپوزیسیون (جامعه ۴) نیز روابط پیچیده‌ای (چه مثبت و چه منفی) دارند.
	\end{itemize}

	\begin{itemize}
		\item \textbf{خوشه مرکزی (جامعه ۰):} در توییتر، خوشه مرکزی با ۲۴۸ عضو، بزرگ‌تر از تلگرام است. این تفاوت می‌تواند به دلیل ماهیت بازتر و عمومی‌تر توییتر باشد که به طیف وسیع‌تری از حساب‌های موافق نظام اجازه فعالیت می‌دهد. اهمیت میانگین این خوشه (۱.۰۹) تقریباً با سایر خوشه‌ها برابری می‌کند، که نشان می‌دهد در توییتر، هسته مرکزی از "وزن" ویژه‌ای نسبت به بدنه اصلی اپوزیسیون برخوردار نیست.
		\item \textbf{خوشه‌های عظیم اپوزیسیون (جوامع ۱ و ۴):} دو خوشه با اندازه‌های ۵۶۳۳ و ۴۴۵۹ عضو، بدنه اصلی مخالفان در توییتر را تشکیل می‌دهند. اندازه غول‌آسای این خوشه‌ها، حکایت از ظرفیت بالای توییتر برای بسیج سیاسی و شکل‌دهی به گفتمان اعتراضی در سطح گسترده دارد [citation:5]. با این حال، وجود دو خوشه مجزا و بزرگ، تأییدی بر شکاف عمیق درون جریان اپوزیسیون است. این شکاف می‌تواند بر سر راهبردها (خشونت‌آمیز یا مسالمت‌آمیز بودن اعتراضات)، رهبری (نقش رضا پهلوی [citation:10]) یا هویت قومی باشد.
		\item \textbf{جامعه ۳:} مشابه تلگرام، یک خرده‌گروه بسیار کوچک (۳ عضوی) در توییتر نیز دیده می‌شود که می‌تواند نشان‌دهنده یک گروه افراطی یا یک هسته بسیار هماهنگ و خاص باشد.
	\end{itemize}
	
	\section{تحلیل تطبیقی و پیش‌بینی آینده روابط خوشه‌ها}
	با ترکیب تحلیل تعادل ساختاری و ریخت‌شناسی خوشه‌ها، می‌توان چشم‌اندازی از آینده روابط درون شبکه‌های اجتماعی ایران ترسیم کرد. سناریوهای زیر محتمل‌تر به نظر می‌رسند:
	
	\begin{enumerate}
		\item \textbf{سناریوی اول: تداوم شکاف و ناپایداری (محتمل‌ترین سناریو)} با توجه به نسبت بالای سه‌گانه‌های نامتعادل و وجود خوشه‌های متعدد و مجزا در اپوزیسیون، به نظر می‌رسد شبکه در وضعیت "ناپایدار پایدار" (Stable Instability) باقی بماند. به این معنا که تنش‌های درونی اپوزیسیون به حدی است که مانع از شکل‌گیری یک ائتلاف منسجم و فراگیر علیه هسته مرکزی می‌شود. هسته مرکزی (به ویژه در تلگرام با اهمیت بالا) از این شکاف‌ها برای تضعیف جریان مخالف بهره‌برداری خواهد کرد. شواهد تاریخی از اعتراضات گذشته، مانند جنبش "زن، زندگی، آزادی" نیز نشان‌دهنده چالش‌های اپوزیسیون در حفظ انسجام در بلندمدت در برابر اقدامات سرکوبگرانه و محدودیت‌های اینترنتی است [citation:3] [citation:4].
		
		\item \textbf{سناریوی دوم: تلاش برای بازآرایی و ائتلاف‌های جدید} فشار ناشی از سرکوب یا یک رویداد سیاسی مهم (مانند تشدید بحران اقتصادی) می‌تواند برخی از خرده‌گروه‌های اپوزیسیون را به سمت ائتلاف‌های جدید سوق دهد. برای مثال، ممکن است خرده‌گروه‌های کرد و بلوچ که خواسته‌های هویتی مشترکی دارند، با وجود رقابت‌های قومی، یک ائتلاف تاکتیکی علیه سیاست‌های متمرکز دولت تشکیل دهند. چنین ائتلافی می‌تواند با "مثبت" کردن روابط منفی پیشین، سه‌گانه‌های نامتعادل را به متعادل تبدیل کرده و ثبات شبکه را به نفع اپوزیسیون تغییر دهد. در این میان، بازیگران خارجی و دیاسپورای ایرانی می‌توانند نقش "پل‌زننده" را میان این گروه‌ها ایفا کنند [citation:6].
		
		\item \textbf{سناریوی سوم: تقویت هسته مرکزی و به حاشیه‌راندن اپوزیسیون} اگر هسته مرکزی بتواند با استفاده از راهبردهای محاسباتی و پروپاگاندا [citation:9]، روایت خود را مسلط کرده و شکاف‌های درون اپوزیسیون را عمیق‌تر کند، ممکن است بخشی از مخالفان میانه‌رو را به سکوت یا حتی همکاری با خود جذب نماید. داده‌های توییتر نشان می‌دهد که اهمیت هسته مرکزی با خوشه‌های بزرگ اپوزیسیون برابری می‌کند، که این نشان‌دهنده توانایی رقابت این هسته در جنگ روایت‌هاست. استفاده از ابزارهای پیشرفته فناوری اطلاعات توسط بازیگران دولتی برای سرکوب و رصد مخالفان، همانطور که در کمپین "RedKitten" مشاهده شد [citation:2]، این سناریو را تقویت می‌کند.
	\end{enumerate}
	
\\
	تحلیل شبکه‌های اجتماعی تلگرام، اینستاگرام و توییتر در جریان اعتراضات دی‌ماه ۱۴۰۴، تصویری از یک جامعه شبکه‌ای شده با قطبی‌شدگی بالا و ناپایداری ساختاری عمیق را به نمایش می‌گذارد. نظریه تعادل ساختاری نشان می‌دهد که این ناپایداری عمدتاً ناشی از شکاف‌های متعدد درون اپوزیسیون (سلطنت‌طلبان، کردها، ترک‌ها، بلوچ‌ها) است که مانع از شکل‌گیری یک جبهه متحد در برابر هسته مرکزی موافق نظام می‌شود. هسته مرکزی نیز به نوبه خود با بهره‌گیری از قدرت تأثیرگذاری بالای خود (به ویژه در تلگرام) و استفاده از تاکتیک‌های محاسباتی، به تعمیق این شکاف‌ها دامن می‌زند. آینده روابط این خوشه‌ها بیش‌ از آن که به سمت یک ائتلاف فراگیر پیش رود، در وضعیت تنش و بازآرایی‌های تاکتیکی باقی خواهد ماند، مگر آن که یک رویداد برون‌زا یا تغییر راهبردی در رهبری اپوزیسیون، پویایی‌های جدیدی را ایجاد کند.

این تحلیل با استفاده از چارچوب کلاسیک «ارزش‌های خبری» (News Values) که توسط گالتونگ و روگه در سال ۱۹۶۵ پایه‌ریزی شد، به بررسی عمیق محتوای شبکه‌های اجتماعی فارسی‌زبان در ایام اعتراضات دی ۱۴۰۴ می‌پردازد . بر اساس این نظریه، رویدادی شانس تبدیل شدن به خبر را دارد که معیارهایی چون «فراوانی» (تناسب با بازه زمانی رسانه)، «آستانه‌ای بودن» (اهمیت و شدت)، «عدم‌ابهام»، «معناداری فرهنگی»، «همخوانی با انتظارات»، «غیرمنتظره بودن»، «تداوم»، و «تنوع» را داشته باشد. همچنین عوامل وابسته به فرهنگ مانند ارجاع به «افراد نخبه»، «ملت‌های نخبه»، «شخصی‌سازی» و «منفی‌گرایی» در برجسته‌سازی یک رویداد مؤثرند .

در این گزارش، ۵ مطلب پربازنشر روزانه در سه پلتفرم تلگرام، اینستاگرام و توییتر (X) در بازه ۱۵ تا ۲۰ دی ۱۴۰۴ بر اساس این معیارها تحلیل می‌شوند. این تحلیل نشان می‌دهد چگونه کنشگران اجتماعی و سیاسی در بسترهای دیجیتال، با بهره‌گیری از این ارزش‌ها، به تولید و انتشار محتوای اعتراضی پرداخته‌اند.

### تحلیل داده‌ها

ارزش‌های خبری در این محتواها به سه دسته اصلی قابل تقسیم هستند: **ارزش‌های مرتبط با نخبگان و قدرت** (Elite-Centric)، **ارزش‌های مرتبط با درگیری و منفی‌گرایی** (Negativity & Conflict)، و **ارزش‌های مرتبط با هویت و نمادهای فرهنگی** (Cultural Proximity & Identity). آنچه در داده‌های زیر مشهود است، غلبه شدید دو دسته اول در اوج اعتراضات است.

### نتیجه‌گیری و تحلیل نهایی

با نگاهی به مجموعه داده‌ها و تحلیل آن‌ها در چارچوب ارزش‌های خبری و علم شبکه، می‌توان به نتایج زیر دست یافت:

1.  **غلبه ارزش «فراد نخبه» (Elite-Centeredness):** در تمامی پلتفرم‌ها، به ویژه توییتر و اینستاگرام، محتوای مرتبط با رضا پهلوی (فراخوان‌ها، پیام‌ها) بالاترین میزان انتشار و تعامل را داشته است. این پدیده با نظریه گالتونگ و روگه در مورد «ارجاع به افراد نخبه» همخوانی کامل دارد . در شرایطی که شبکه‌های رسمی قدرت دچار بی�اعتباری شده‌اند، شبکه‌های اجتماعی یک «نخبه جایگزین» برای رهبری روایت خلق می‌کنند .

2.  **پلتفرمی شدن ارزش‌ها (Platformization of News Values):** هر پلتفرم ارزش‌های خبری را به شکلی متفاوت بازتولید کرد:
*   **توییتر:** بستری برای ارزش «اقتدار و رهبری» (Authority). پیام‌ها مستقیم، رسمی و با هدف هماهنگی ( Coordination ) صادر می‌شدند .
*   **اینستاگرام:** بستری برای ارزش «هویت و تعلق» (Identity). تصاویر، نمادها (شیر و خورشید) و هشتگ‌های عاطفی (#جاویدشاه) در این پلتفرم برجسته‌تر بودند.
*   **تلگرام:** بستری برای ارزش «سرعت و شایعه» (Speed & Rumor). اخبار تأییدنشده، ویدیوهای خام از درگیری‌ها و محتوای توطئه‌آمیز در این بستر سریع‌ترین انتشار را داشت .

3.  **تغییر ارزش‌ها در طول زمان (Temporal Shift):**
*   در **روزهای ابتدایی (۱۵-۱۶ دی)** ، ارزش‌های «اقتصادی» و «نارضایتی عمومی» (یارانه، گرانی) برجسته بود.
*   با **اوج‌گیری اعتراضات (۱۷-۱۸ دی)** ، ارزش «درگیری و خشونت» (Negativity) و «فراد نخبه» به اوج خود رسید.
*   در **روزهای پایانی (۱۹-۲۰ دی)** ، ارزش‌های «نمادین و هویتی» (Symbolism) مانند برافراشته شدن پرچم‌ها و همچنین «واکنش‌های بین‌المللی» (ترامپ) به چرخه اخبار اضافه شدند .

4.  **بازیگران کلیدی و توزیع قدرت در شبکه:** علم شبکه نشان می‌دهد که توزیع قدرت در این شبکه‌های اجتماعی کاملاً متمرکز بوده است. حساب‌های وابسته به رضا پهلوی به عنوان «مراکز قدرت» (Hubs) عمل کرده و پیام را تعیین می‌کردند. در مقابل، کانال‌های خبری مانند `vahidonline` در تلگرام نقش «گره‌های توزیع‌کننده» (Distributor Nodes) را ایفا کرده و پیام را از پلتفرمی به پلتفرم دیگر منتقل می‌کردند.

در نهایت، داده‌های ارائه‌شده تصویری از یک جنبش اجتماعی شبکه‌شده را ترسیم می‌کند که در آن ارزش‌های کلاسیک خبری (مانند نخبه‌محوری و منفی‌گرایی) نه تنها کمرنگ نشده، بلکه در بسترهای جدید دیجیتال با شدتی بیشتر بازتولید شده‌اند. شبکه‌های اجتماعی در دی ۱۴۰۴ به صحنه‌ای برای «بازار رقابتی روایت‌ها» تبدیل شدند که در آن، موفق‌ترین محتواها آنهایی بودند که بهترین ترکیب را از «سرعت»، «احساس» و «اقتدار» ارائه می‌دادند.
